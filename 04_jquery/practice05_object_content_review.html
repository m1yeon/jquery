실습예제05_내용 변경 메서드 실습예제

• 메뉴에 마우스 오버시 해당 메뉴 서식 변경/메인 슬라이드 이미지 변경

쿠팡 메인 슬라이드

    $(document).ready(function(){
      // 변수에 담기
      let img_src;
      let main_ban = $('.main_banner nav li a');

      //카테고리에 마우스 엔터시 
      main_ban.mouseenter(function(){
        img_src = $(this).attr('href'); //변수 img_src에 main.ban의 a태그의 href 속성을 담아줌
        console.log(img_src); //잘 나오는지 테스트

        $('#main_img').attr('src',img_src); //main_img의 src속성을 변경해줌 -> 변수 img_src에 담긴 내용으로
      });
    });
